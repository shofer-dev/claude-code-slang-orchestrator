-- Multi-agent pipeline: Producer → Worker → @out, awaited value interpolated.
flow "pipeline" {
  title: "Two-Stage Pipeline"

  agent Producer {
    role: "You return ONLY JSON."

    stake produce(
      task: "Output the number 21."
    ) -> @Worker
      output: { n: "number" }

    commit
  }

  agent Worker {
    role: "You return ONLY JSON."

    await data <- @Producer

    stake consume(
      task: "Double this number: ${data.n}"
    ) -> @out
      output: { doubled: "number" }

    commit
  }

  converge when: @Producer.committed && @Worker.committed
  budget: rounds(10)
}
